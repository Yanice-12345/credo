let currentQuestionIndex = 0; const questionDisplay = document.getElementById("questionDisplay"); const answerInput = document.getElementById("answer"); const message = document.getElementById("message"); const btnPlay = document.getElementById("btnPlay"); const btnSlow = document.getElementById("btnSlow"); const btnFast = document.getElementById("btnFast"); const btnRepeat = document.getElementById("btnRepeat"); const btnSkip = document.getElementById("btnSkip"); const btnBack = document.getElementById("btnBack"); let speechRate = 1; // Fonction pour remplacer les symboles mathématiques par des mots function replaceOperators(text) { return text .replace(/÷/g, "diviser par") // Remplace ÷ par "diviser par" .replace(/×/g, "fois") .replace(/-/g, "moins") .replace(/\+/g, "plus"); } function speakQuestion(text, rate = speechRate) { if (!('speechSynthesis' in window)) { message.textContent = "La synthèse vocale n'est pas disponible."; return; } speechSynthesis.cancel(); // Annule toute lecture en cours const utterance = new SpeechSynthesisUtterance(text); utterance.lang = "fr-FR"; utterance.rate = rate; speechSynthesis.speak(utterance); } function speakMessage(text) { speakQuestion(text, speechRate); } function explainMathCalculation(question) { let explanation = ''; if (question === "6 × 2 + 4 ÷ 2") { explanation = "D'abord, on effectue la multiplication : 6 × 2 = 12. Ensuite, on fait la division : 4 ÷ 2 = 2. Puis, on additionne : 12 + 2 = 14."; } // Ajoute d'autres explications pour chaque question ici. return explanation; } function playQuestion() { const current = questions[currentQuestionIndex]; const questionText = replaceOperators(current.question); // Remplacer les opérateurs dans la question questionDisplay.textContent = current.question; // Lire la question à haute voix speakQuestion(`Résolvez cette question : ${questionText}`); message.textContent = "Entrez la réponse correcte."; message.style.color = "#fff"; answerInput.value = ""; answerInput.focus(); // Annonce de l'explication de la correction const explanation = explainMathCalculation(current.question); if (explanation) { speakMessage(explanation); } } function playQuestionSlow() { speechRate = Math.max(0.5, speechRate - 0.25); playQuestion(); } function playQuestionFast() { speechRate = Math.min(2, speechRate + 0.25); playQuestion(); } function repeatQuestion() { playQuestion(); } function checkAnswer() { const userAnswer = parseInt(answerInput.value.trim(), 10); const correctAnswer = questions[currentQuestionIndex].answer; if (userAnswer === correctAnswer) { message.textContent = "✅ Correct ! Bien joué."; message.style.color = "#00e676"; speakMessage("Bravo, vous avez trouvé la bonne réponse !"); setTimeout(() => { nextQuestion(); }, 1500); } else { message.textContent = `❌ Incorrect. La bonne réponse était : ${correctAnswer}`; message.style.color = "#ff5252"; speakMessage(`Désolé, la réponse correcte était ${correctAnswer}.`); } } function nextQuestion() { currentQuestionIndex++; if (currentQuestionIndex >= questions.length) { message.textContent = "🎉 Bravo, vous avez complété la liste !"; questionDisplay.textContent = ""; answerInput.value = ""; answerInput.disabled = true; btnPlay.disabled = true; btnSlow.disabled = true; btnFast.disabled = true; btnRepeat.disabled = true; btnSkip.disabled = true; btnBack.disabled = true; speakMessage("Bravo, vous avez terminé le jeu !"); } else { playQuestion(); } } btnBack.addEventListener("click", () => { if (currentQuestionIndex > 0) { currentQuestionIndex--; message.textContent = ""; playQuestion(); } }); btnPlay.addEventListener("click", playQuestion); btnSlow.addEventListener("click", playQuestionSlow); btnFast.addEventListener("click", playQuestionFast); btnRepeat.addEventListener("click", repeatQuestion); btnSkip.addEventListener("click", nextQuestion); answerInput.addEventListener("keydown", (e) => { if (e.key === "Enter") { checkAnswer(); } }); // Démarrage automatique window.addEventListener("load", playQuestion);
